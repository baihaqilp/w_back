SELECT MAX(id) FROM addresses;
SELECT nextval('addresses_id_seq');
BEGIN;
LOCK TABLE your_table IN EXCLUSIVE MODE;
SELECT setval('addresses_id_seq', COALESCE((SELECT MAX(id)+1 FROM addresses), 1), false);
COMMIT;
